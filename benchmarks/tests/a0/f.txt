f1(n)
	if (n = null) return
	f1(n.l)
	f1(n.r)
	if (n.isLeaf)
		n.v = 1
	else
		n.v = n.l.v + n.r.v

f2(n)
	if (n = null) return
	if (n.isLeaf)
		n.x = n.tmpx
		// dumb
		n.l.tmpx = -9999
	else
		n.x = n.v
		n.l.tmpx = n.x + 2
	n.y = n.v * 2
	f2(n.l)
	f2(n.r)
